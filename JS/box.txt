  <script>
    // Usuarios y contraseñas (esto debe ser manejado de manera segura en el backend)
    const usuarios = [
      { username: "admin", password: "admin123" },
      { username: "user", password: "user123" }
    ];

    // Validar inicio de sesión
    function iniciarSesion() {
      const username = document.getElementById("username").value;
      const password = document.getElementById("password").value;

      // Comprobar si las credenciales son correctas
      const usuario = usuarios.find(u => u.username === username && u.password === password);

      
      if (usuario) {
        alert("Inicio de sesión exitoso!");
        document.getElementById("login").style.display = "none";
        document.getElementById("productos").style.display = "flex";
        cargarProductos(); // Cargar productos después de login
      } else {
        alert("Usuario o contraseña incorrectos.");
      }
    }

    // Cargar productos
    function cargarProductos() {
      fetch('/productos')
        .then(res => res.json())
        .then(data => {
          const contenedor = document.getElementById('productos');
          contenedor.innerHTML = "";

          data.forEach(producto => {
            const div = document.createElement('div');
            div.classList.add('producto');
            div.innerHTML = `
              <img src="${producto.imagen}" alt="${producto.nombre}">
              <h3>${producto.nombre}</h3>
              <p class="precio">Precio actual: $${producto.precio}</p>
              <input type="number" id="precio-${producto.codigo}" value="${producto.precio}">
              <button class="btn" onclick="actualizarPrecio('${producto.codigo}')">Actualizar</button>
              <button class="btn btn-danger" onclick="eliminarProducto('${producto.codigo}')">Eliminar</button>
            `;
            contenedor.appendChild(div);
          });
        });
    }

    // Actualizar precio
    function actualizarPrecio(codigo) {
      const nuevoPrecio = document.getElementById(`precio-${codigo}`).value;
      fetch('/actualizar', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({codigo, nuevoPrecio: parseFloat(nuevoPrecio)}),
      })
      .then(res => res.text())
      .then(msg => { alert(msg); cargarProductos(); });
    }

    // Agregar nuevo producto
    function agregarProducto() {
      const nombre = document.getElementById("nuevoNombre").value;
      const codigo = document.getElementById("nuevoCodigo").value;
      const precio = parseFloat(document.getElementById("nuevoPrecio").value);
      const imagen = document.getElementById("nuevaImagen").value;

      fetch('/agregar', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ nombre, codigo, precio, imagen })
      })
      .then(res => res.text())
      .then(msg => { alert(msg); cargarProductos(); });
    }

    // Eliminar producto
    function eliminarProducto(codigo) {
      if (!confirm("¿Seguro que deseas eliminar este producto?")) return;

      fetch('/eliminar', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ codigo })
      })
      .then(res => res.text())
      .then(msg => { alert(msg); cargarProductos(); });
    }
